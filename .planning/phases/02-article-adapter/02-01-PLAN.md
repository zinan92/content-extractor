       ```python
       def _compute_word_count(text: str) -> int:
           if not text:

           cjk_chars = len(re.findall(r'[\u4e00-\u9fff\u3400-\u4dbf]', text))
           latin_words = len(re.findall(r'[a-zA-Z]+', text))
           return cjk_chars + latin_words
       ```
